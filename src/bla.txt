#include <Arduino.h>

#define PIN_UD D2
#define PIN_INC D3
#define POT_DOWN false
#define POT_UP true
void potCmd(boolean up){
  digitalWrite(PIN_UD, up);
  digitalWrite(PIN_INC, HIGH);
  delayMicroseconds(1e3);
  digitalWrite(PIN_INC, LOW);
}

void setup() {
  pinMode(PIN_INC, OUTPUT);
  pinMode(PIN_UD, OUTPUT);
  //pinMode(A0, INPUT);
  for (int i=0; i<100; i++) {
    potCmd(POT_UP);
    delayMicroseconds(150);
  }
}
void loop() {
  for (int i=0; i<100; i++){
    potCmd(POT_DOWN);
    delay(1500);
  }
  for (int i=0; i<100; i++){
    potCmd(POT_UP);
    delay(1500);
  }
}
